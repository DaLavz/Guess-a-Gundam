const SHEET_NAME = "Leaderboard"; function doGet(e) { var action = e.parameter.action || "get"; if (action === "get") return getLeaderboard(); return respond({ error: "Unknown action" }); } function doPost(e) { return submitScore(e); } function getOrCreateSheet() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow(["name", "mc", "type", "zoom", "impostor", "pilot", "updated"]); sheet.getRange(1, 1, 1, 7).setFontWeight("bold"); } return sheet; } function getLeaderboard() { var sheet = getOrCreateSheet(); var data = sheet.getDataRange().getValues(); if (data.length <= 1) return respond({ rows: [] }); var rows = []; for (var i = 1; i < data.length; i++) { var row = data[i]; if (!row[0]) continue; rows.push({ name: row[0], mc: row[1], type: row[2], zoom: row[3], impostor: row[4], pilot: row[5], updated: row[6] }); } // Sort by total numeric score descending rows.sort(function(a, b) { return extractTotal(b) - extractTotal(a); }); return respond({ rows: rows }); } function submitScore(e) { var body; try { body = JSON.parse(e.postData.contents); } catch (err) { return respond({ error: "Invalid JSON" }); } var name = String(body.name || "").trim().substring(0, 24); if (!name) return respond({ error: "Name required" }); // Store display strings as-is (e.g. "8/10", "3/5", "42 pts") var mc = String(body.mc || ""); var type = String(body.type || ""); var zoom = String(body.zoom || ""); var impostor = String(body.impostor || ""); var pilot = String(body.pilot || ""); var sheet = getOrCreateSheet(); var data = sheet.getDataRange().getValues(); var rowIndex = -1; for (var i = 1; i < data.length; i++) { if (String(data[i][0]).toLowerCase() === name.toLowerCase()) { rowIndex = i + 1; break; } } var updated = new Date().toISOString(); if (rowIndex > 0) { var ex = data[rowIndex - 1]; // Keep whichever display value represents a higher numeric score sheet.getRange(rowIndex, 1, 1, 7).setValues([[ name, pickBetter(ex[1], mc), pickBetter(ex[2], type), pickBetter(ex[3], zoom), pickBetter(ex[4], impostor), pickBetter(ex[5], pilot), updated ]]); } else { sheet.appendRow([name, mc, type, zoom, impostor, pilot, updated]); } return respond({ ok: true }); } // Extract the leading number from a display string like "8/10", "3/5", "42 pts" function extractNum(str) { if (!str) return 0; var m = String(str).match(/^(\d+)/); return m ? parseInt(m[1], 10) : 0; } // Pick whichever display string has the higher leading number function pickBetter(existing, incoming) { if (!existing || existing === "") return incoming; if (!incoming || incoming === "") return existing; return extractNum(incoming) > extractNum(existing) ? incoming : existing; } // Sum up numeric parts of all modes for sorting function extractTotal(row) { return extractNum(row.mc) + extractNum(row.type) + extractNum(row.zoom) + extractNum(row.impostor) + extractNum(row.pilot); } function respond(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
